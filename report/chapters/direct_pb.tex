\section{Direct problem}

\subsection{Variational formulation}

The boundary conditions are:
\begin{itemize}
  \item Left and right boundaries ($\Gamma_{\text{side}}$): homogeneous Neumann condition $\partial_n T = 0$ (insulated walls).
  \item Top boundary ($\Gamma_{\text{top}}$): Dirichlet condition $T = T_c = 343\,\mathrm{K}$.
  \item Bottom boundary ($\Gamma_{\text{bottom}}$): Robin condition $\partial_n T + h(T - T_e) = 0$ where $T_e = 293\, \mathrm{K}$ and $h = 1$ (door).\
\end{itemize}

The temperature T inside the oven is given by the stationary heat equation:
\begin{equation}\label{base_eq}
  -\mathrm{div}(k\nabla T) = f
\end{equation}
where $k$ equals 1 inside $S$ and 10 otherwise, and $f$ is the heat source term that as the form $f = \sum_{i=1}^6 \alpha_i \mathbb{1}_{C_i}$.

We introduce a lifting function $\tilde{T} \in H^1(\Omega)$ such that $\tilde{T}|_{\Gamma_{\text{top}}} = T_c$. Then, we write $T = \tilde{T} + T_0$, where $T_0$ satisfies the homogeneous Dirichlet condition on $\Gamma_{\text{top}}$.

By multiplying~\cref{base_eq} by a test function $v \in V = \{v \in H^1(\Omega) \mid v|_{\Gamma_{\text{top}}} = 0\}$ and integrating by parts, we obtain:
\begin{equation}
\int_\Omega k\,\nabla T \cdot \nabla v \dd x - \int_{\partial\Omega} k\,\partial_n T\, v \dd\sigma = \int_\Omega f\,v \dd x.
\end{equation}
By substituting the boundary conditions (zero Neumann on $\Gamma_{\text{side}}$, Robin on $\Gamma_{\text{bottom}}$, and $v=0$ on $\Gamma_{\text{top}}$) and adding the contribution of the lifting function, we arrive at the variational formulation:

\begin{equation}\label{eq:VF}
  \boxed{
  \text{Find } T_0 \in V \text{ such that } a(T_0,v) = l(v), \quad \forall v \in V
  }
\end{equation}where:
\begin{equation}
a(T_0,v) = \int_\Omega k\,\nabla T_0 \cdot \nabla v \dd x + \int_{\Gamma_{\text{bottom}}} k h\,T_0\,v \dd\sigma,
\end{equation}
\begin{equation}
l(v) = \int_\Omega f\,v \dd x + \int_{\Gamma_{\text{bottom}}} k h\,(T_e - \tilde{T})\,v \dd\sigma - \int_\Omega k\,\nabla \tilde{T} \cdot \nabla v \dd x
\end{equation}

\paragraph{Analysis (Lax-Milgram Theorem)}
Let $V = \{v \in H^1(\Omega) \mid v = 0 \text{ on } \Gamma_{\text{top}}\}$ be equipped with the $H^1$-norm $\|v\|_{H^1}^2 = \|v\|_{L^2}^2 + \|\nabla v\|_{L^2}^2$. Recall that the thermal conductivity $k \in L^\infty(\Omega)$ satisfies $1 \le k(\mathbf{x}) \le 10$ a.e.\,in $\Omega$, and $h = 1$.

To apply the Lax-Milgram theorem, we must verify three hypotheses:

\begin{enumerate}
  \item \textbf{Continuity of the bilinear form $a(\cdot, \cdot)$:}
  For all $u, v \in V$, using the triangle inequality and the upper bound of $k$, we have:
  \begin{equation*}
      |a(u,v)| \le 10 \displaystyle\left\lvert\int_\Omega \nabla u \cdot \nabla v \dd x \right \rvert + 10 \displaystyle\left\lvert\int_{\Gamma_{\text{bottom}}} uv \dd\sigma \right\rvert.
  \end{equation*}
  By applying Cauchy-Schwarz, and using the Trace inequality for the right integral, we obtain:
  \begin{equation*}
      |a(u,v)| \le 10 \|\nabla u\|_{L^2} \|\nabla v\|_{L^2} + 10 C_{\text{tr}}^2 \|u\|_{H^1} \|v\|_{H^1}.
  \end{equation*}
  Since $\|\nabla v\|_{L^2} \le \|v\|_{H^1}$, there exists a constant $M > 0$ such that $|a(u,v)| \le M \|u\|_{H^1} \|v\|_{H^1}$. Thus, $a(\cdot, \cdot)$ is continuous.

  \item \textbf{Coercivity of $a(\cdot, \cdot)$:}
  For all $v \in V$, since $k \ge 1$ and $k h v^2 \ge 0$, we have:
  \begin{equation*}
      a(v,v) = \int_\Omega k |\nabla v|^2 \dd x + \int_{\Gamma_{\text{bottom}}} k h v^2 \dd\sigma \ge \int_\Omega |\nabla v|^2 \dd x = \|\nabla v\|_{L^2}^2.
  \end{equation*}
  Since $\Omega$ is connected and the Dirichlet boundary $\Gamma_{\text{top}}$ has a strictly positive measure, the Poincaré inequality holds on $V$. 
  
  Thus, there exists a constant $C_P > 0$ such that $\|v\|_{L^2(\Omega)} \le C_P \|\nabla v\|_{L^2(\Omega)}$ for all $v \in V$.
  
  Using the definition of the $H^1$-norm ($\|v\|_{H^1}^2 = \|v\|_{L^2}^2 + \|\nabla v\|_{L^2}^2$), this implies that there exists $\alpha > 0$ such that:
  \begin{equation*}
      a(v,v) \ge \|\nabla v\|_{L^2}^2 \ge \alpha \|v\|_{H^1}^2.
  \end{equation*}
  Thus, the bilinear form $a(\cdot, \cdot)$ is coercive on $V$.

  \item \textbf{Continuity of the linear form $l(\cdot)$:}
  For all $v \in V$, we use the triangle inequality and the bounds of $k$:
  \begin{equation*}
      |l(v)| \le \displaystyle\left\lvert \int_\Omega f v \dd x \right \rvert+ 10 \displaystyle\left\lvert \int_{\Gamma_{\text{bottom}}} (T_e - \tilde{T}) v \dd\sigma \right \rvert+ 10 \displaystyle\left\lvert \int_\Omega  \nabla \tilde{T} \cdot \nabla v \dd x \right\rvert.
  \end{equation*}
  Applying the Cauchy-Schwarz inequality and the Trace inequality:
  \begin{equation*}
      |l(v)| \le \|f\|_{L^2} \|v\|_{L^2} + 10 \|T_e - \tilde{T}\|_{L^2(\Gamma_{\text{bottom}})} C_{\text{tr}} \|v\|_{H^1} + 10 \|\nabla \tilde{T}\|_{L^2} \|\nabla v\|_{L^2}.
  \end{equation*}
  Since $\tilde{T}$ is a fixed function in $H^1(\Omega)$ and $f \in L^2(\Omega)$, all terms multiplying the norms of $v$ are bounded constants. Hence, there exists $C > 0$ such that:
  \begin{equation*}
      |l(v)| \le C \|v\|_{H^1}.
  \end{equation*}
  Thus, $l(\cdot)$ is continuous.
\end{enumerate}

\textbf{Conclusion:} Since $V$ is a Hilbert space, $a(\cdot, \cdot)$ is a continuous and coercive bilinear form, and $l(\cdot)$ is a continuous linear form, the Lax-Milgram theorem guarantees the existence and uniqueness of the solution $T_0 \in V$.

\subsection{Finite element approximation}

We approximate $V$ by a finite-dimensional space $V_h \subset V$, based on a regular triangular mesh $\mathcal{T}_h$ of $\Omega$ using $\mathbb{P}_1$ finite elements. Let ${(\varphi_j)}_{j=1}^N$ be the nodal basis functions associated with the nodes that are not on the Dirichlet boundary (i.e., strictly vanishing on $\Gamma_{\text{top}}$).

We approximate the unknown $T_0$ by $T_{0,h} = \sum_{j=1}^N U_j\,\varphi_j \in V_h$, where the $U_j$ are the unknown temperatures at the nodes. Similarly, the continuous lifting function is discretized as $\tilde{T}_h$, which takes the fixed value $T_c$ at the nodes located on $\Gamma_{\text{top}}$ and vanishes at all other nodes.

By substituting $T_{0,h}$ into the continuous variational formulation of~\cref{eq:VF} and choosing the test functions $v = \varphi_i$ for $i=1, \dots, N$, we obtain the linear system:
\begin{equation}
\mathbf{A}\,\mathbf{U} = \mathbf{b},
\end{equation}
where the elements of the matrix $\mathbf{A}$ and the right-hand side vector $\mathbf{b}$ are directly deduced from the bilinear and linear forms:
\begin{equation}
A_{ij} = a(\varphi_j, \varphi_i) = \int_\Omega k\,\nabla\varphi_j\cdot\nabla\varphi_i\dd x + \int_{\Gamma_{\text{bottom}}} k h\,\varphi_j\,\varphi_i\dd\sigma,
\end{equation}
\begin{equation}
b_i = l(\varphi_i) = \int_\Omega f\,\varphi_i\dd x + \int_{\Gamma_{\text{bottom}}} k h\,(T_e - \tilde{T}_h)\,\varphi_i\dd\sigma - \int_\Omega k\,\nabla \tilde{T}_h \cdot \nabla \varphi_i \dd x.
\end{equation}

The matrix $\mathbf{A}$ is symmetric, positive-definite, and sparse. Once the system is solved for the vector $\mathbf{U}$, the full approximate temperature distribution in the oven is reconstructed by adding back the discrete lifting function:
\begin{equation}
T_h(\mathbf{x}) = \sum_{j=1}^N U_j\,\varphi_j(\mathbf{x}) + \tilde{T}_h(\mathbf{x}).
\end{equation}

\textit{Remark: In our numerical implementation with FreeFem++, the construction of $\tilde{T}_h$ and the transfer of the lifting terms to the right-hand side $\mathbf{b}$ are handled automatically, so we directly solve for the full temperature $T_h$ without explicitly separating the lifting function. We implement the equation on FreeFem++ as:
\begin{equation}
\int_\Omega k\,\nabla T \cdot \nabla v \dd x + \int_{\Gamma_{\text{bottom}}} k h\,T\,v \dd\sigma
= \int_\Omega f\,v \dd x + \int_{\Gamma_{\text{bottom}}} k h T_e\,v \dd\sigma
\end{equation}
}

\subsection{FreeFem++ implementation and comments}

The numerical simulation (see \texttt{directPb.edp}) of the direct problem yields the temperature distribution inside the oven for a uniform set of source intensities ($\alpha_i = 10^4$), see \cref{fig:four_direct}.

\begin{figure}[H]
    \centering
    \includegraphics[width=\textwidth]{fig/directPb.eps}
    \caption{Temperature distribution inside the oven for the direct problem ($\alpha_i = 10^4$).}\label{fig:four_direct}
\end{figure}

From the obtained temperature map (\cref{fig:four_direct}), several key physical behaviors can be observed.

Firstly, the top wall boundary condition is correctly enforced, as the temperature at the top boundary is uniformly equal to $T_c$.
The lower boundary displays significant heat loss towards the external environment at temperature $T_e$, which is perfectly consistent with the Robin boundary condition.
On the left and right insulated boundaries (homogeneous Neumann conditions), the isotherms are strictly perpendicular to the walls. This graphically confirms that the normal heat flux is zero.

Then, the region $S$ is not heated uniformly, since the top of the piece to cook is warmer than the bottom. So, we don't have a good temperature distribution for cooking, which motivates the optimization problem in~\cref{sec:opti_pb}, in order to have different source intensities $\alpha_i$ that can yield a more uniform temperature distribution in $S$.

\subsection{Description of the oven warm up phase}

To model the heating phase of the oven from an initial cold state, we introduce the time variable $t \in [0, T]$. The temperature distribution $T(\mathbf{x}, t)$ is governed by the transient heat equation, which includes the thermal inertia term $\rho c$ (volumetric heat capacity):
\begin{equation}
  \rho c \frac{\partial T}{\partial t} - \mathrm{div}(k\nabla T) = f \quad \text{in } \Omega \times [0, T],
\end{equation}
subject to the same boundary conditions as the stationary problem. Assuming normalized physical properties for the air inside the oven, we set $\rho c = 1$. Initially, the oven is at room temperature:
\begin{equation}
    T(\mathbf{x}, 0) = T_e \quad \text{in } \Omega.
\end{equation}

We introduce a time step $\delta t > 0$ and denote $t_n = n \delta t$. To ensure unconditional numerical stability, we use the first-order implicit Euler scheme, approximating the time derivative by $\partial_t T^{n+1} \approx \frac{T^{n+1} - T^n}{\delta t}$.

Following the stationary case procedure, we introduce the bilinear form $m(u,v) = \int_\Omega u \, v \dd x$. By isolating the unknown terms at time $t_{n+1}$ on the left-hand side, we obtain the fully discrete variational formulation at each time step:
\begin{equation} \label{eq:VF_time}
  \boxed{
  \begin{aligned}
    &\text{For } n \ge 0, \text{ find } T^{n+1}_h \in V_h \text{ such that } \forall v_h \in V_h, \\
    &m(T^{n+1}_h, v_h) + \delta t \, a(T^{n+1}_h, v_h) = m(T^n_h, v_h) + \delta t \, l(v_h),
  \end{aligned}
  }
\end{equation}
where $a(\cdot, \cdot)$ and $l(\cdot)$ are the forms defined in \cref{eq:VF}. As $t \to +\infty$, this transient solution naturally converges toward the stationary solution.

At the final time step (here $T = 0.5$), we obtain (see \texttt{transientPb.edp}) a temperature distribution that is close to the stationary solution, which validates the implementation of the transient scheme (see \cref{fig:transientPb}). 

\begin{figure}[H]
    \centering
    \includegraphics[width=\textwidth]{fig/transientPb.eps}
    \caption{Temperature distribution inside the oven for the transient direct problem ($\alpha_i = 10^4$) at the final time step.}\label{fig:transientPb}
\end{figure}

The warm-up phase can be visualized as a sequence of temperature maps at different time steps, showing how the oven heats up from the initial cold state to the steady-state distribution.
